\begin{zremark}
  关于相等关系的证明，常用办法。
\end{zremark}

\begin{itemize}
  \item (1) 通过中间变量实现。如：$x = z, y = z \Longrightarrow x = y$。
  \item (2) 通过任意$\epsilon > 0$，有$|x - y| < \epsilon \Longrightarrow x = y$。
\end{itemize}
